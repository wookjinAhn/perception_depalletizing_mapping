## Find catkin macros and libraries

        roscpp
        sensor_msgs
        tf
        depalletizing_mapping_base
        depalletizing_mapping_core
        depalletizing_mapping_ros
        depalletizing_mapping_msgs
)

###################################

## CATKIN_DEPENDS: catkin_packages dependent projects also need

        INCLUDE_DIRS
            include
        LIBRARIES
            ${PROJECT_NAME}
        CATKIN_DEPENDS
            roscpp
            sensor_msgs
            tf
            depalletizing_mapping_base
            depalletizing_mapping_core
            depalletizing_mapping_ros

        DEPENDS
)
